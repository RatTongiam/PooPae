    <script>

        let myChart = null;


        document.getElementById('csvFileInput').addEventListener('change', function(e) {

            const file = e.target.files[0];

            if (!file) return;


            const reader = new FileReader();

            reader.onload = function(event) {

                processData(event.target.result);

            };

            reader.readAsText(file);

        });


        function processData(text) {

            const samplingRate = parseFloat(document.getElementById('samplingRate').value);

            const contactThreshold = parseFloat(document.getElementById('contactThreshold').value);

            const stabilityWindowSec = parseFloat(document.getElementById('stabilityWindow').value);

            const windowSize = Math.floor(stabilityWindowSec * samplingRate);


            const lines = text.split(/\r?\n/);

            if (lines.length < 2) return;


            // ตรวจสอบหัวข้อคอลัมน์จากแถวแรก

            const headers = lines[0].split(';');

            const idxForce = headers.indexOf('2 Newton');

            const idxSwayX = headers.indexOf('2 sway X');

            const idxSwayY = headers.indexOf('2 sway Y');

            const idxVelocity = headers.indexOf('2 velocity');

            const idxTime = headers.indexOf('timestamp');


            if (idxForce === -1) {

                alert("โครงสร้างไฟล์ไม่ถูกต้อง: ไม่พบคอลัมน์ '2 Newton'");

                return;

            }


            let timestamps = [];

            let forceData = [];

            let swayXData = [];

            let swayYData = [];

            let velocityData = [];


            // ทำการคลี่ข้อมูลลูปอ่านค่าจากแถว

            for (let i = 1; i < lines.length; i++) {

                if (!lines[i].trim()) continue;

                const cols = lines[i].split(';');

                if (cols.length < headers.length) continue;


                timestamps.push(parseFloat(cols[idxTime]));

                forceData.push(parseFloat(cols[idxForce]));

                swayXData.push(parseFloat(cols[idxSwayX]));

                swayYData.push(parseFloat(cols[idxSwayY]));

                velocityData.push(parseFloat(cols[idxVelocity]));

            }


            const timeSecs = timestamps.map(t => t / samplingRate);


            // ---- 1. คำนวณพารามิเตอร์พื้นฐาน ----

            const peakForce = Math.max(...forceData);

            const maxVelocity = Math.max(...velocityData.map(Math.abs));

            

            const minSwayX = Math.min(...swayXData);

            const maxSwayX = Math.max(...swayXData);

            const minSwayY = Math.min(...swayYData);

            const maxSwayY = Math.max(...swayYData);


            // ---- 2. คำนวณตรรกะวิเคราะห์ SL-TTS ----

            let idxContact = forceData.findIndex(f => f >= contactThreshold);

            let tContactSec = null;

            let tStabilitySec = null;

            let slTtsSec = null;

            let bw = 0;

            let upperCombined = 0;

            let lowerCombined = 0;


            if (idxContact !== -1) {

                tContactSec = timeSecs[idxContact];

                

                // คำนวณหา Body Weight จาก 100 จุดสุดท้าย

                const last100 = forceData.slice(-100);

                bw = last100.reduce((a, b) => a + b, 0) / last100.length;

                upperCombined = bw + (0.05 * bw);

                lowerCombined = bw - (0.05 * bw);


                // ค้นหา Onset of Stability ตามเกณฑ์หน้าต่างต่อเนื่อง 1 วินาที

                for (let i = idxContact; i <= forceData.length - windowSize; i++) {

                    const window = forceData.slice(i, i + windowSize);

                    const isStable = window.every(f => f >= lowerCombined && f <= upperCombined);

                    if (isStable) {

                        tStabilitySec = timeSecs[i];

                        slTtsSec = tStabilitySec - tContactSec;

                        break;

                    }

                }

            }


            // ---- 3. อัปเดตข้อมูลขึ้นหน้าจอระบบ UI ----

            document.getElementById('metricPeakForce').innerText = `${peakForce.toFixed(2)} N`;

            document.getElementById('metricBW').innerText = bw > 0 ? `Body Weight อ้างอิง: ${bw.toFixed(2)} N` : `Body Weight อ้างอิง: -`;

            

            if (slTtsSec !== null) {

                document.getElementById('metricTTS').innerText = `${slTtsSec.toFixed(3)} วินาที`;

                document.getElementById('metricContactTime').innerText = `Initial Contact ที่: ${tContactSec.toFixed(3)} วินาที\nStability Onset ที่: ${tStabilitySec.toFixed(3)} วินาที`;

            } else {

                document.getElementById('metricTTS').innerText = `ไม่เข้าเกณฑ์คงที่`;

                document.getElementById('metricContactTime').innerText = idxContact !== -1 ? `Initial Contact ที่: ${tContactSec.toFixed(3)} วินาที` : `ไม่พบจุดปะทะแรง 10N`;

            }


            document.getElementById('metricSwayX').innerText = `↔️ Max Range Sway X: ${(maxSwayX - minSwayX).toFixed(2)}`;

            document.getElementById('metricSwayY').innerText = `↕️ Max Range Sway Y: ${(maxSwayY - minSwayY).toFixed(2)}`;

            document.getElementById('metricVelocity').innerText = `💨 Peak Velocity: ${maxVelocity.toFixed(6)} m/s`;


            document.getElementById('resultsPanel').classList.remove('hidden');

            document.getElementById('chartContainer').classList.remove('hidden');


            // ---- 4. สร้างและเรนเดอร์กราฟด้วย Chart.js ----

            if (myChart) myChart.destroy();


            const ctx = document.getElementById('forceChart').getContext('2d');

            

            // สร้างเส้นระดับแนวนอนสำหรับขอบเขตความเสถียร

            const annotations = [];

            if (bw > 0) {

                annotations.push({ type: 'line', yMin: bw, yMax: bw, borderColor: 'green', borderWidth: 2, label: { content: 'BW', enabled: true } });

            }


            myChart = new Chart(ctx, {

                type: 'line',

                data: {

                    labels: timeSecs,

                    datasets: [{

                        label: 'Vertical Ground Reaction Force (N)',

                        data: forceData,

                        borderColor: '#7f7f7f',

                        borderWidth: 1.5,

                        pointRadius: 0,

                        fill: false

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    scales: {

                        x: { type: 'linear', title: { display: true, text: 'Time (Seconds)' } },

                        y: { title: { display: true, text: 'Force (Newtons)' } }

                    }

                }

            });

        }

    </script>
